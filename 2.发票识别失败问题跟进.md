@Override
	public ApiResult invoiceCheckOCR(OcrCheckReq req) {
		// ocr识别缓存，如果存在，则从缓存取值
		String cacheKey = SysConsts.getCanonicalCacheKey("ocr:invoice:check:{0}", req.getInvoiceNumber());
		ApiResult result = cacheService.get(cacheKey, ApiResult.class);

		if (result == null) {
			result = new ApiResult();
			try {
				Map<String, String> params = Maps.newHashMap();
				params.put("billing_date", req.getBillingDate());
				params.put("invoice_number", req.getInvoiceNumber());
				params.put("invoice_code", req.getInvoiceCode());
				if (req.getTotalAmount() != null) {
					params.put("total_amount", req.getTotalAmount().toString());
				}
				String checkCode = req.getCheckCode();
				log.info("该发票检查码为{}", checkCode);
				if (StringUtils.isNotBlank(checkCode) && checkCode.length() >= 6) {
					String code = checkCode.substring(checkCode.length() - 6);
					log.info("该发票检查码后六位为{}", code);
					params.put("check_code", code);
				} else {
					params.put("check_code", checkCode);
				}


				log.info("OCR验证请求参数:{}", JSON.toJSONString(params));
				result = gtnApiUtil.send(params, "/data_check/v1/ocr/invoice_check");
				// 存缓存
				this.initCache(cacheKey, result, null);
			} catch (Exception e) {
				log.error("", e);
				result.setErrorMsg(e.getMessage());
			}
		}
		log.info("OCR验证响应结果:{}", JSON.toJSONString(result));

		return result;
	}

# 复盘
这里边几个问题
1.缓存的使用，使用redis缓存通常需要过期时间的（配置项等特殊场景除外），这里的缓存没有设置过期时间，经验来讲最多设置为6-24个小时。
2.请求三方后，失败的结果一般不存缓存，这里存了，从性能角度可以接受这么做，但是没有过期时间，导致这张发票由于第一次失败，后续就一直失败了
3.请求三方发票验真为什么失败 ？ 返回报文给了明确的失败 code ，由于频次极低，不再进一步跟进
